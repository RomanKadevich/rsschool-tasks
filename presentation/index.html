Example of Using State in a Component


					import React, { useState } from 'react';

					const Counter = () => {
					const [count, setCount] = useState(0);

					const increment = () => {
						setCount(count + 1);
					};

					return (
						<div>
							<h3>>Counter:{count}</h3>
							<button onClick={increment}>Increment</button>
						</div>
					);
					};

					export default Counter;
